\chapter{Exerciții suplimentare EDP1}

Exercițiile de mai jos au fost în lista pentru parțial din anul 2019-2020.

\begin{enumerate}

\item Determinați liniile și suprafața de câmp corespunzătoare câmpului vectorial:
  \[
    \vec{v} = (x^3 + 3xy^2)\vec{i} + 2y^3 \vec{j} + 2y^2z \vec{k}.
  \]

\item Determinați suprafața de câmp a câmpului vectorial:
  \[
    \vec{v} = x\vec{i} + y\vec{j} + (z - x^2 - y^2) \vec{k},
  \]
  care conține curba dată de:
  \[
    \gamma: \begin{cases} y &= -2 \\ z &= x - x^2 \end{cases}.
  \]

\item Să se afle liniile de câmp asociate câmpului vectorial:
  \[
    \vec{v}=(xz)\vec{i}-(yz)\vec{j}+(x^{2}+y^{2})\vec{k}.
  \]

\item Să se afle liniile de câmp asociate câmpului vectorial:
  \[
    \vec{v}=(3z-4y)\vec{i}+(4x-2z)\vec{j}+(2y-3x)\vec{k}.
  \]

\item Să se afle liniile de câmp asociate câmpului vectorial:
  \[
    \vec{v} = e^x \cdot \vec{i} + y^2 \vec{j} + e^x y \vec{k}.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    2x \dfrac{\partial u}{\partial x} - y\dfrac{\partial u}{\partial y} + %
    (4xy^2 - 2z) \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    (2x - 3y)\dfrac{\partial u}{\partial x} + (3x - z) \dfrac{\partial u}{\partial y} + %
    (y - 2x) \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    (y - z) \dfrac{\partial u}{\partial x} + (z - x) \dfrac{\partial u}{\partial y} + %
    (x - y) \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    (xz + y) \dfrac{\partial u}{\partial x} + (x + yz) \dfrac{\partial u}{\partial y} + %
    (1 - z^2) \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    xy^2 + \dfrac{\partial u}{\partial x} + x^2 y \dfrac{\partial u}{\partial y} + %
    z(x^2 + y^2) \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați soluția generală a ecuației cu derivate parțiale de
  ordinul întâi:
  \[
    xy \dfrac{\partial u}{\partial x} + (x^2 + y^2) \dfrac{\partial u}{\partial y} + %
    yz \dfrac{\partial u}{\partial z} = 0.
  \]

\item Aflați liniile de câmp asociate câmpului vectorial:
  \[
    \vec{v} = (4z - 5y) \vec{i} + (5x - 3z)\vec{j} + (3y - 4x) \vec{k}.
  \]

\item Determinați liniile și suprafața de câmp corespunzătoare
  câmpului vectorial:
  \[
    \vec{v} = (z - y)^2 \vec{i} + xz \vec{j} + xy \vec{k}.
  \]

\item Determinați liniile și suprafața de câmp corespunzătoare
  câmpului vectorial:
  \[
    \vec{v} = (x^3 + 3xy^2) \vec{i} + 2y^3 \vec{j} + 2y^2z \vec{k}.
  \]

\item Determinați suprafața de câmp corespunzătoare câmpului vectorial:
  \[
    \vec{v} = x(y + z) \vec{i} + z(z - y) \vec{j} + y(y - z) \vec{k},
  \]
  care conține curba:
  \[
    \gamma : \begin{cases}
      x &= 4 \\
      y^2 - z^2 &= 4
    \end{cases}.
  \]

\item Determinați suprafața de câmp corespunzătoare câmpului vectorial:
  \[
    \vec{v} = (x^2 + y^2) \vec{i} + 2xy \vec{j} + xz \vec{k},
  \]
  care conține curba:
  \[
    \gamma: \begin{cases}
      z &= 1 \\
      x^2 - y^2 &= 1 
    \end{cases}.
  \]
  Identificați tipul suprafeței.

\item Determinați suprafețele de câmp asociate câmpului vectorial:
  \[
    \vec{v} = xz \vec{i} + yz \vec{j} + (x + y) \cdot z \vec{k}.
  \]

\item Determinați suprafețele de câmp asociate câmpului vectorial:
  \[
    \vec{v} = xz^2 \vec{i} + y(x^2 + z^2)\vec{j} + x^2 z \vec{k}.
  \]

\item Determinați suprafețele de câmp asociate câmpului vectorial:
  \[
    \vec{v} = (x^2 + y^2) \vec{i} + 2xy \vec{j} - z^2 \vec{k}.
  \]

\item Determinați suprafețele de câmp asociate câmpului vectorial:
  \[
    \vec{v} = 2xz\vec{i} + 2zy \vec{j} + (z^2 - x^2 - y^2) \vec{k},
  \]
  care conțin cercul dat de:
  \[
    \gamma: \begin{cases}
      z &= 0 \\
      x^2 + y^2 - 2x &= 0
    \end{cases}.
  \]
\end{enumerate}
