\chapter{Recapitulare analiză complexă}

\begin{enumerate}[(1)]
    \item Determinați partea reală sau partea imaginară a funcției $ f = P + iQ $ olomorfă (dacă există) pentru:
    \begin{enumerate}[(a)]
        \item $ P(x, y) = x^2 - y^2 $ și $ f(0) = 0 $;
        \item $ Q(x, y) = \ln(x^2 + y^2) + x - 2y $;
        \item $ P(x, y) = xy + x + 2y $ și $ f(2i) = -1 + 5i $;
        \item $ P(x, y) = 4xy^3 - 4x^3y + x $ și $ f(1 + i) = 5 + 4i $;
        \item $ P(x, y) = 3x^2y + 2x^2 - y^3 - 2y^2 $;
        \item $ Q(x, y) = \dfrac{x}{x^2 + y^2} $;
        \item $ Q(x, y) = \dfrac{y}{x^2 + y^2} $.
    \end{enumerate}
    \item Determinați $ a \in \RR $ astfel încît $ P(x, y) = 2xy + 3x^2 - ay^2 + 2ax + 3y - 4 $
    să fie partea reală a unei funcții olomorfe. Determinați funcția $ f(z) $ pentru care $ f(i) = 1 $.
    \item Calculați integralele complexe:
    \begin{enumerate}[(a)]
        \item $ \ds\int_{|z| = 2} \dfrac{\exp(z)}{(z - 1)(z - 3i)} dz $;
        \item $ \ds\int_{|z| = 1} \dfrac{\sin(z)}{z^2 - 5z + 4} dz $;
        \item $ \ds\int_{|z - 1| = 2} \dfrac{z}{z^2 + 2} dz $;
        \item $ \ds\int_{|z - i| = 1} \dfrac{\exp(z)}{z^2 + 1} dz $;
        \item $ \ds\int_{|z - 1| = 1} \dfrac{z + 2}{(z - 1)^2(z + 2)} dz $;
        \item $ \ds\int_{|z + i| = 2} \dfrac{\exp(2z)}{z^2(z^2 + 1)(z^2 + 2)} dz $;
        \item $ \ds\int_{|z - 2i| = 1} \dfrac{\exp{3z}}{z^2 + 10} + \dfrac{z^2 + 1}{z^3(z + 2)} dz $.
    \end{enumerate}
    \item Folosind funcții complexe, calculați integralele trigonometrice:
    \begin{enumerate}[(a)]
        \item $ \ds\int_0^{2\pi} \dfrac{1}{(2 + \cos t)^2} dt $;
        \item $ \ds\int_0^{2\pi} \dfrac{dt}{2 + \cos t} dt $;
        \item $ \ds\int_0^{2 \pi} \dfrac{dt}{4 + \cos t} dt $;
        \item $ \ds\int_0^{2\pi} \dfrac{dt}{5 + 3 \cos t} dt $;
        \item $ \ds\int_0^{2\pi} \dfrac{dt}{3 + 2 \cos^2 t} dt $.
    \end{enumerate}
\end{enumerate}
