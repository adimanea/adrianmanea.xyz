\chapter{Recapitulare pentru examen}

\section*{Set 1}

\indent\indent 1. Rezolvați următoarele ecuații diferențiale de ordin superior:
\begin{enumerate}[(a)]
\item $ 4(x + 1)^2 y^{(2)} + y = 0, \quad x > -1 $;
\item $ y^{(4)} - 3y^{(3)} + 3y^{(2)} - y^{(1)} = 0 $;
\item $ y^{(3)} = \sin x + \cos x $.
\end{enumerate}

\vspace{1cm}

2. Rezolvați ecuația cu derivate parțiale:
\[
  2yz_x + 3x^2 z_y + 6x^2 y = 0,
\]
funcția necunoscută fiind $ z = z(x, y) $.

\vspace{1cm}

3. Determinați suprafața de cîmp a cîmpului vectorial:
\[
  \vec{V} = 2xz\vec{i} + 2yz\vec{j} + (z^2 - x^2 - y^2) \vec{k},
\]
care conține cercul dat de $ z = 0 $ și $ x^2 + y^2 - 2x = 0 $.

\vspace{1cm}

4. Fie cîmpul scalar:
\[
  f(x, y, z) = 3x^2 - 5xy + 2x^2z^2.
\]

Calculați $ \nabla \times (\nabla f) $ în punctul $ A(1, 0, 1) $.

\section*{Set 2}

\indent\indent 1. Rezolvați ecuațiile de ordin superior:
\begin{enumerate}[(a)]
\item $ xy^{(2)} + 3y^{(1)} = 1 $;
\item $ y^{(2)} - y^{(1)} - 2y = 3e^{2x} $.
\end{enumerate}

\vspace{1cm}

2. Rezolvați sistemul diferențial:
\[
  \begin{cases}
    x' &= x + 3y \\
    y' &= -x + 5y - 2e^t
  \end{cases},
\]
cu funcțiile necunoscute $ x = x(t), y = y(t) $ și cu condițiile inițiale
$ x(0) = 3 $ și $ y(0) = 1 $.

\vspace{1cm}

3. Fie cîmpul vectorial:
\[
  \vec{v} = (zx - y) \vec{i} + (x - zy) \vec{j} + (x^2 - y^2) \vec{k}.
\]

Determinați liniile sale de cîmp și suprafețele de cîmp care conțin curba
dată de $ xy = 1 $ și $ z = 1 $.

\vspace{1cm}

4. Fie cîmpul vectorial:
\[
  \vec{v} = (3x, -2y + z, x^2 + z^2).
\]

Calculați $ \nabla \times \vec{v} $ în punctul $ A(-1, 1, 1) $.
