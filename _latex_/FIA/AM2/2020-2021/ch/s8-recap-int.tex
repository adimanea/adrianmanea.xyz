\chapter{Recapitulare formule integrale}

Folosind, eventual, \textbf{formula Green-Riemann}, calculați integralele:
\begin{enumerate}[(a)]
    \item $ \ds\int_\gamma xydx + xy^2 dy$, unde $ \gamma: x^2 - 4x + y^2 = 0 $;
    \item $ \ds\int_\gamma y^2 dx + (x + y) dy $, unde $ \gamma: x^2 + y^2 + 4y = 0 $;
    \item $ \ds\int_\gamma (x + y) dx + (x^2 - y) dy $, unde $ \gamma $ este pătratul cu vîrfurile în $ A(1,1), B(3,1), C(3,3), D(1,3) $;
    \item $ \ds\int_\gamma (x^2 - y) dx + (2x + 3y) dy $, unde $ \gamma $ este triunghiul cu vîrfurile în $ A(1, 1), B(1, 3), C(4,1) $;
    \item $ \ds\int_\gamma 2xydx + x^2 dy $, unde $ \gamma = \{ (x, y) \in \RR^2 \mid x^2 + y^2 = 4, x \leq 0 \} $;
    \item $ \ds\int_\gamma xdx + 2x^2y dy $, unde $ \gamma = \{ (x, y) \in \RR^2 \mid x^2 + y^2 = 9, x, y \leq 0 \} $;
    \item $ \ds\int_\gamma 2xydx + (x - y) dy $, unde $ \gamma = \{ (x, y) \in \RR^2 \mid x^2 + y^2 = 4, x \geq y \} $.
\end{enumerate}

\vspace{0.3cm}

Folosind, eventual, \textbf{formula Gauss-Ostrogradski}, calculați fluxul cîmpului vectorial $ V $
prin suprafața $ \Sigma $, unde:
\begin{enumerate}[(a)]
    \item $ V = (2x, y, 2xy), \Sigma = x^2 + y^2 + z^2 = 9 $;
    \item $ V = (x^2, y, 3z - 1), \Sigma = \{ (x, y, z) \in \RR^3 \mid z = x^2 + y^2, z \leq 2 \} \cup \{ (x, y, z) \in \RR^3 \mid x^2 + y^2 = 2, z = 2 \} $;
    \item $ V = (xz, z, z^2 - 1), \Sigma = x^2 + y^2 + z^2 = 4, x \geq 0 $;
    \item $ V = (x^2, xz, xz), \Sigma = z = 2x^2 + 2y^2, z \leq 6 $;
    \item $ V = (yz, x^2, y^2), \Sigma = y = x^2 + z^2, y \leq 2 $;
    \item $ V = (x + z, y + x, z + y), \Sigma = 2x = y^2 + z^2, x \leq 1 $.
\end{enumerate}

\vspace{0.3cm}

Folosind, eventual, \textbf{formula Stokes}, calculați circulația cîmpului vectorial $ V $,
prin curba de intersecție a suprafețelor $ \Sigma_1 $ și $ \Sigma_2 $ date:
\begin{enumerate}[(a)]
    \item $ V = (x, y, z), \Sigma_1 = x^2 + y^2 + z^2 = 4, \Sigma_2 = x = 1 $;
    \item $ V = (x - y, z - y, x - z), \Sigma_1 = z = x^2 + y^2, z \leq 3, \Sigma_2 = x^2 + y^2 + z^2 = 1 $;
    \item $ V = (2xy, x, x), \Sigma_1 = y = x^2 + z^2, y \leq 1, \Sigma_2 = x^2 + y^2 + z^2 = 1 $.
\end{enumerate}

\vspace{0.3cm}

\textbf{Bonus:} Calculați ariile suprafețelor:
\begin{enumerate}[(a)]
    \item $ \Sigma = 2y = x^2 + z^2, y \leq 2 $;
    \item $ \Sigma = x = 3y^2 + 3z^2, x \leq 1 $;
    \item $ \Sigma = x^2 + y^2 + z^2 = 4, x \geq 0, y \geq 0 $.
\end{enumerate}
